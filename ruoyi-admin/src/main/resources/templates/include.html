DROP TABLE IF EXISTS `iot_environl`; CREATE TABLE `iot_environl` ( `Ambient temperature` varchar(255) NOT NULL COMMENT '环境温度', `ambient humidity` varchar(255) NOT NULL COMMENT '环境湿度', `wind speed` Varchar(255) DEFAULT NULL COMMENT '风速', `wind direction` varchar(255) CHARACTER SET utf8 COLLATE utf8_general_ci DEFAULT NULL COMMENT '风向', `Light intensity` varchar(255) DEFAULT NULL COMMENT '光照强度', `atmospheric pressure` varchar(255) DEFAULT NULL COMMENT '大气压强', `PM2.5`varchar(255) DEFAULT NULL COMMENT 'PM2.5浓度', `ppm` varchar(255) DEFAULT NULL COMMENT '二氧化碳浓度', `Device status` varchar(255) DEFAULT NULL COMMENT '设备状态（0-关，1-开）' );
